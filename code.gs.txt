// ---------- DO POST ----------
function doPost(e) {
  try {
    const rawData = e.parameter.data;
    if (!rawData) throw new Error('Data tidak ditemukan');
    const data = JSON.parse(rawData);
    const action = data.action;
    if (action === 'generateQuestions') {
      return generateQuestions(data);
    } else if (action === 'evaluateAnswer') {
      return evaluateAnswer(data);
    } else {
      return error('Unknown action');
    }
  } catch (err) {
    return error(err.toString());
  }
}

// ---------- DO GET ----------
function doGet(e) {
  const action = e.parameter.action;
  if (action === 'getResults') {
    return getResults();
  }
  return error('Invalid action');
}

// ---------- GENERATE BACAAAN (dengan fallback) ----------
function generateQuestions(data) {
  const level = data.level;
  const configs = {
    1: { wordRange: '150-250', targetWords: 200, timeRead: 300, timeWrite: 180, count: 2 },
    2: { wordRange: '400-500', targetWords: 450, timeRead: 600, timeWrite: 300, count: 1 },
    3: { wordRange: '700-900', targetWords: 800, timeRead: 900, timeWrite: 420, count: 1 }
  };
  const cfg = configs[level];
  if (!cfg) return error('Level tidak valid');

  let questions = null;
  try {
    const prompt = `Buatkan ${cfg.count} bacaan/cerita pendek untuk anak SMP level ${level}. 
    Setiap bacaan harus memiliki panjang sekitar ${cfg.targetWords} kata (antara ${cfg.wordRange} kata). 
    Tema bisa sains, kehidupan sehari-hari, alam, sejarah, atau lainnya. 
    Bacaan harus informatif, menarik, dan memiliki alur cerita yang jelas.
    Output harus berupa JSON array dengan ${cfg.count} objek, masing-masing memiliki properti "text" yang berisi teks bacaan lengkap.
    Contoh format: [{"text": "Isi bacaan pertama..."}, {"text": "Isi bacaan kedua..."}]
    Hanya outputkan JSON array, tanpa markdown atau teks tambahan.`;

    const response = callGemini(prompt);
    // REVISI: Langsung melakukan JSON.parse karena respon dijamin JSON murni oleh opsi responseMimeType
    const parsed = JSON.parse(response); 
    
    if (Array.isArray(parsed) && parsed.length === cfg.count) {
      let valid = true;
      for (let item of parsed) {
        if (!item.text || typeof item.text !== 'string') {
          valid = false;
          break;
        }
        const wordCount = item.text.split(/\s+/).length;
        const minWords = Math.floor(cfg.targetWords * 0.7);
        const maxWords = Math.floor(cfg.targetWords * 1.3);
        if (wordCount < minWords || wordCount > maxWords) {
          valid = false;
          break;
        }
      }
      if (valid) {
        questions = parsed.map((item) => ({
          text: item.text,
          timeRead: cfg.timeRead,
          timeWrite: cfg.timeWrite
        }));
      }
    }
  } catch (e) {
    // fallback
  }

  if (!questions) {
    const fallback = FALLBACK_READINGS[level];
    if (fallback && fallback.length >= cfg.count) {
      questions = fallback.slice(0, cfg.count).map((item) => ({
        text: item.text,
        timeRead: cfg.timeRead,
        timeWrite: cfg.timeWrite
      }));
    } else {
      questions = Array.from({ length: cfg.count }, (_, i) => ({
        text: `Bacaan #${i+1}: (Fallback) ${fallback?.[0]?.text || 'Gagal mengambil bacaan. Silakan coba lagi.'}`,
        timeRead: cfg.timeRead,
        timeWrite: cfg.timeWrite
      }));
    }
  }

  return success({ questions });
}

// ---------- EVALUASI JAWABAN (dengan ide pokok) ----------
function evaluateAnswer(data) {
  const { level, question, answer, name, index } = data;
  const prompt = `Berikut adalah sebuah bacaan dan ringkasan/ide pokok yang ditulis oleh siswa.

  Bacaan:
  ${question}

  Ringkasan/Ide Pokok siswa:
  ${answer}

  Tugasmu: 
  1. Bandingkan ringkasan siswa dengan isi bacaan. 
  2. Nilai seberapa baik siswa menangkap ide pokok dan informasi penting dari bacaan.
  3. Berikan skor 1-10 (10 = sangat baik, menangkap semua ide pokok dengan tepat).
  4. Berikan komentar singkat (2-3 kalimat) yang membangun untuk siswa.
  5. Tuliskan ide pokok yang SEHARUSNYA dari bacaan tersebut (dalam 1-2 kalimat).

  Output dalam format JSON: 
  {"score": angka (1-10), "comment": "komentar", "expected_idea": "ide pokok yang diharapkan"}. 
  Hanya output JSON.`;

  let result = { score: 5, comment: 'Analisis tidak tersedia.', expected_idea: 'Tidak tersedia' };
  let errorMsg = '';

  for (let attempt = 1; attempt <= 2; attempt++) {
    try {
      const response = callGemini(prompt);
      // REVISI: Langsung melakukan JSON.parse karena respon dijamin JSON murni oleh opsi responseMimeType
      const parsed = JSON.parse(response);
      if (parsed.score !== undefined && parsed.comment && parsed.expected_idea) {
        result = parsed;
        break;
      } else {
        // fallback ekstrak jika properti tidak lengkap
        const scoreMatch = response.match(/score["']?\s*:\s*(\d+)/i);
        const commentMatch = response.match(/comment["']?\s*:\s*["']([^"']*)["']/i);
        const ideaMatch = response.match(/expected_idea["']?\s*:\s*["']([^"']*)["']/i);
        if (scoreMatch) result.score = parseInt(scoreMatch[1]);
        if (commentMatch) result.comment = commentMatch[1];
        if (ideaMatch) result.expected_idea = ideaMatch[1];
        if (scoreMatch || commentMatch || ideaMatch) break;
        errorMsg = 'Format JSON tidak sesuai';
      }
    } catch (e) {
      errorMsg = e.toString();
    }
  }

  // Simpan ke sheet dengan kolom Ide Pokok
  saveToSheet(name, level, question, answer, result.score, result.comment, result.expected_idea);
  return success({ score: result.score, comment: result.comment, expected_idea: result.expected_idea });
}

// ---------- SIMPAN KE SHEET (tambah kolom Ide Pokok) ----------
function saveToSheet(name, level, question, answer, score, comment, expected_idea) {
  const ss = SpreadsheetApp.openById(SHEET_ID);
  let sheet = ss.getSheetByName('Data');
  if (!sheet) {
    ss.insertSheet('Data');
    sheet = ss.getSheetByName('Data');
    // Header dengan 8 kolom
    sheet.appendRow(['Waktu', 'Nama', 'Level', 'Bacaan', 'Ringkasan Siswa', 'Skor', 'Komentar', 'Ide Pokok']);
  } else {
    // Cek apakah kolom Ide Pokok ada, jika tidak tambahkan
    const headers = sheet.getRange(1, 1, 1, sheet.getLastColumn()).getValues()[0];
    if (headers.indexOf('Ide Pokok') === -1) {
      sheet.getRange(1, sheet.getLastColumn() + 1).setValue('Ide Pokok');
    }
  }
  const timestamp = new Date().toLocaleString('id-ID', { timeZone: 'Asia/Jakarta' });
  sheet.appendRow([timestamp, name, level, question, answer, score, comment, expected_idea || '']);
}

// ---------- GET HASIL (termasuk ide pokok) ----------
function getResults() {
  const sheet = SpreadsheetApp.openById(SHEET_ID).getSheetByName('Data');
  if (!sheet) {
    return success({ results: [] });
  }
  const data = sheet.getDataRange().getValues();
  if (data.length < 2) return success({ results: [] });
  
  const headers = data[0];
  const waktuIdx = headers.indexOf('Waktu');
  const nameIdx = headers.indexOf('Nama');
  const levelIdx = headers.indexOf('Level');
  const scoreIdx = headers.indexOf('Skor');
  const ideaIdx = headers.indexOf('Ide Pokok');
  const results = [];
  for (let i = 1; i < data.length; i++) {
    const row = data[i];
    results.push({
      waktu: waktuIdx > -1 ? row[waktuIdx] : '-',
      name: nameIdx > -1 ? row[nameIdx] : 'Anonim',
      level: levelIdx > -1 ? row[levelIdx] : '-',
      score: scoreIdx > -1 ? row[scoreIdx] : 0,
      expected_idea: ideaIdx > -1 ? row[ideaIdx] : ''
    });
  }
  return success({ results });
}

// ---------- PANGGIL GEMINI ----------
function callGemini(prompt) {
  const payload = {
    contents: [
      {
        parts: [{ text: prompt }]
      }
    ],
    // REVISI: Memaksa model Gemini memberikan kembalian objek JSON murni tanpa markdown pembungkus
    generationConfig: {
      responseMimeType: "application/json"
    }
  };
  const options = {
    method: 'post',
    headers: { 'Content-Type': 'application/json' },
    payload: JSON.stringify(payload),
    muteHttpExceptions: true
  };
  const response = UrlFetchApp.fetch(GEMINI_API_URL, options);
  const data = JSON.parse(response.getContentText());
  console.log('Gemini response:', data);
  if (data.candidates && data.candidates.length > 0) {
    const text = data.candidates[0].content.parts[0].text;
    if (text) return text;
  }
  throw new Error('Respons Gemini tidak valid: ' + JSON.stringify(data));
}

function success(data) {
  return ContentService.createTextOutput(JSON.stringify({ status: 'success', ...data }))
    .setMimeType(ContentService.MimeType.JSON);
}

function error(msg) {
  return ContentService.createTextOutput(JSON.stringify({ status: 'error', message: msg }))
    .setMimeType(ContentService.MimeType.JSON);
}
